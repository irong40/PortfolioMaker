    ```python
    def submit_to_mipmap(photo_dir, output_dir, preset, progress_callback=None):
        """Submit photos to MipMap for Gaussian Splat generation.
        Returns (working_dir, result_dict) on success, (None, error_msg) on failure.
        """
        mipmap_settings = preset.get("mipmap_settings", {})
        working_dir = Path(output_dir) / "mipmap_work"

        result = run_mipmap_pipeline(
            photo_dir=photo_dir,
            working_dir=working_dir,
            progress_callback=progress_callback,
            resolution_level=mipmap_settings.get("resolution_level", 3),
            mesh_decimate_ratio=mipmap_settings.get("mesh_decimate_ratio", 0.5),
        )

        if result.get("returncode", -1) != 0:
            return None, f"MipMap failed (exit code {result.get('returncode')})"

        # Copy outputs to portfolio folder
        copied = copy_splat_outputs(working_dir, output_dir)
        return str(working_dir), copied
    ```
